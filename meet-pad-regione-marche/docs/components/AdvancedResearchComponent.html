import {
    Component,
    Output,
    EventEmitter
} from '@angular/core';
import { FormGroup, FormBuilder } from '@angular/forms';

import { ErrorMessage } from 'ngx-eng-toolkit';
import { I18nService } from 'eng-core-lib';
import { Observable, Subject, concat, of } from 'rxjs';

import {
    UtilityService,
    ComboBox,
    LoaderService,
    SectionLoading
} from '@app/core';
import { map, switchMap, debounceTime, takeUntil, distinctUntilChanged, tap } from 'rxjs/operators';
import { SearchAdvancedCriteria } from '@features/private/search/models/search-advanced-criteria.model';
import { AutoUnsubscribe } from '@app/shared/components/auto-unsubscribe/auto-unsubscribe.class';

@Component({
    // tslint:disable-next-line:component-selector
    selector: 'advanced-research',
    templateUrl: './advanced-research.component.html',
    styleUrls: ['./advanced-research.component.scss']
})
export class AdvancedResearchComponent extends AutoUnsubscribe {
    // tslint:disable-next-line:no-output-on-prefix
    @Output() onNewSearch = new EventEmitter<any>();

    form: FormGroup;
    searchCriteria: SearchAdvancedCriteria;

    // radio
    // stateRadioOptions: Observable<RadioOption[]>;
    // typeRadioOptions: Observable<RadioOption[]>;

    stateRadioOptions: Observable<ComboBox[]>;
    typeRadioOptions: Observable<ComboBox[]>;

    provinceOptions: Observable<
        ComboBox[]
    > = this.utilityService.getProvincies();
    cityOptions: Observable<ComboBox[]>;

    advancedSearchLoading$ = this.loaderService.getLoading$(
        SectionLoading.ADVANCED_SEARCH
    );

    applicantLabel: string = this.i18nService.translate(
        'SEARCH.ADVANCED_RESEARCH.INPUT_LABELS.APPLICANT'
    );

    applicantItems$: Observable<ComboBox[]>;

    applicantTypeahead$ = new Subject<string>();

    applicantLoading = false;

    applicantNotFoundText: string="";

    constructor(
        private fb: FormBuilder,
        private i18nService: I18nService,
        private utilityService: UtilityService,
        private loaderService: LoaderService
    ) {
        super();
        this.applicantItems$ = concat(
            of([]),
            this.applicantTypeahead$
                .pipe(
                    takeUntil(this.destroy$),
                    debounceTime(200),
                    distinctUntilChanged(),
                    tap(() => this.applicantLoading = true),
                    switchMap((search: string) => {
                        if(search && search.length > 2) {
                            this.applicantNotFoundText = this.i18nService.translate('CONFERENCE.AUTOCOMPLETE.ITEMS_NOT_FOUND');
                            this.applicantLoading = false;
                            return this.utilityService.getApplicantList(search);
                        }

                        this.applicantNotFoundText = this.i18nService.translate('CONFERENCE.AUTOCOMPLETE.INSERT_MIN_CHARACTERS');
                        this.applicantLoading = false;
                        return of([]);
                    })
                )
        );

        this._init();
    }

    private _init(): void {
        this.searchCriteria = new SearchAdvancedCriteria();

        this.cityOptions = this.utilityService.getCities(
            this.searchCriteria.province.key
        );
        // this.advancedSearchLoading$.subscribe(res => {
        //     console.log(res);
        // });
        // radio
        // this.typeRadioOptions = this.utilityService.getTypes().pipe(
        //     map((v: RadioOption[]) => {
        //         return [new RadioOption(
        //             this.i18nService.translate('SEARCH.ADVANCED_RESEARCH.RADIO_TYPE.ALL'), ''
        //         )].concat(v);
        //     })
        // );
        this.typeRadioOptions = this.utilityService.getConferenceTypes().pipe(
            map((v: ComboBox[]) => {
                return [
                    {
                        key: 'all',
                        value: this.i18nService.translate(
                            'SEARCH.ADVANCED_RESEARCH.RADIO_TYPE.ALL'
                        )
                    } as ComboBox
                ].concat(v);
            })
        );

        // radio
        // this.stateRadioOptions = this.utilityService.getStates().pipe(
        //     map((v: RadioOption[]) => {
        //         return [new RadioOption(
        //             this.i18nService.translate('SEARCH.ADVANCED_RESEARCH.RADIO_STATE.ALL'), ''
        //         )].concat(v);
        //     })
        // );

        this.stateRadioOptions = this.utilityService.getStates().pipe(
            map((v: ComboBox[]) => {
                return [
                    {
                        key: 'all',
                        value: this.i18nService.translate(
                            'SEARCH.ADVANCED_RESEARCH.RADIO_STATE.ALL'
                        )
                    } as ComboBox
                ].concat(v);
            })
        );

        // stateRadioOptions
        this.form = this.fb.group({
            state: [this.searchCriteria.state, []],
            conferenceType: [this.searchCriteria.conferenceType, []],
            requestReference: [this.searchCriteria.requestReference, []],
            fiscalCode: [this.searchCriteria.fiscalCode, []],
            province: [this.searchCriteria.province, []],
            city: [this.searchCriteria.city, []]
        });

        this.form
            .get('province')
            .valueChanges.pipe(takeUntil(this.destroy$))
            .subscribe((province: ComboBox) => {
                if (province) {
                    this.cityOptions = this.utilityService.getCities(
                        province.key
                    );
                }
            });
    }

    reset(): void {
        this.form.get('fiscalCode').reset();
        this.form.get('requestReference').reset();
    }

    search(): void {
        this.onNewSearch.emit(new SearchAdvancedCriteria(this.form.value));
    }

    fillForm(form: FormGroup, model: any) {
        return null;
    }

    extractDataToSubmit(form: FormGroup) {
        return form.value;
    }

    submitComplete(model: any) {
        alert('Tutto okay');
    }

    submitError(error: ErrorMessage) {
        alert('Errore');
    }

    pageName(): string {
        return 'Advanced research';
    }
}
