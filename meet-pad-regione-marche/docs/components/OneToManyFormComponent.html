import { Component, Input, EventEmitter, Output } from '@angular/core';
import { FormGroup } from '@angular/forms';

import { Observable } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

import { LoggerService } from 'eng-core-lib';

import { FormFieldGroup } from '@features/private/conference';

import { AutoUnsubscribe } from '@app/shared/components/auto-unsubscribe/auto-unsubscribe.class';

declare var $: any;

@Component({
    selector: 'app-one-to-many-form',
    templateUrl: './one-to-many-form.component.html',
    styleUrls: ['./one-to-many-form.component.scss']
})
export class OneToManyFormComponent extends AutoUnsubscribe {
    @Input('group') group: FormFieldGroup;
    @Input('groupName') groupName: string;
    @Input('form') form: FormGroup;
    @Input('errorLabels') errorLabels: any;
    @Output('removeControl') removeControl: EventEmitter<
        string
    > = new EventEmitter();

    @Input() addRowFn: (groupName: string) => Observable<void>;
    @Input() submitRowFn: (groupName: string) => Observable<any>;

    open = false;
    insertActive = false;
    loading = false;

    constructor(private loggerService: LoggerService) {
        super();
    }

    isPanel(): boolean {
        return this.group.panel;
    }

    isPanelBodyOpen(): boolean {
        return this.open || !!this.group.listMany.length;
    }

    collapsePanel(mode: string, delay: number) {
        // con effetto transizione
        // setTimeout(() => {
        //     $('.collapse').collapse(mode);
        // }, delay);

        // senza effetto transizione
        setTimeout(() => {
            this.open = mode === 'show' ? true : false;
        }, delay);
    }

    openPanel(delay: number = 0): void {
        this.collapsePanel('show', delay);
    }

    closePanel(delay: number = 0): void {
        this.collapsePanel('hide', delay);
    }

    toogleForm() {
        if (this.form) {
            //  aperto va chiuso
            this.removeControl.emit(this.groupName);
            if (!this.group.listMany.length) {
                // il pannello va chiuso se non ci sono utenti in tabella
                this.closePanel();
            }
        } else {
            //  chiuso va aperto
            this._addRow();
        }
    }

    _addRow(): void {
        this.insertActive = true;
        this.addRowFn(this.groupName)
            .pipe(takeUntil(this.destroy$))
            .subscribe(() => {
                this.openPanel();
            });
    }

    onSubmit(object): void {
        this.loading = true;
        this.submitRowFn(this.groupName)
            .pipe(takeUntil(this.destroy$))
            .subscribe(() => {
                this.insertActive = false;
                this.loading = false;
                object.onComplete();
                if (!this.group.listMany.length) {
                    this.closePanel();
                }
            });
    }

    resetForm(): void {
        this.form.reset();
    }

    haveHead(): boolean {
        return this.isPanel() && (this.group.panelHead as boolean);
    }

    getKeys(map): {}[] {
        return Array.from(map.keys());
    }
}

<div [ngClass]="{'panel panel-default': isPanel()}" class="panel-with-manual-transiction">

    <div
