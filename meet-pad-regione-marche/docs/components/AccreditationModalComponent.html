import {
    Component,
    Input,
    ViewChild,
    OnInit,
    Output,
    EventEmitter
} from '@angular/core';
import { FormControl, Validators } from '@angular/forms';
import { Observable } from 'rxjs';

import {
    ErrorMessage,
    ModalComponent,
    ErrorLabelConstants
} from 'ngx-eng-toolkit';

import { UtilityService } from '@app/core';

import { environment } from '@env/environment';

import { FormFieldGroup } from '@features/private/conference/interfaces/form-field-group.interface';

import { FormStoreService } from '@app/features/private/conference/components/form/services/form-store.service';
import { Accreditation } from '@features/private/conference/models/accreditation';

@Component({
    selector: 'app-accreditation-modal',
    templateUrl: './accreditation-modal.component.html',
    styleUrls: ['./accreditation-modal.component.scss']
})
export class AccreditationModalComponent implements OnInit {
    @ViewChild('modal') modal: ModalComponent;

    @Input('accreditationModel') accreditationModel: Accreditation;

    @Input('saveAccreditation') saveAccreditation: (
        data: { accreditation: Accreditation }
    ) => Observable<Accreditation>;
    @Input('saveComplete') saveComplete: (response: Accreditation) => void;

    @Input('requiredFile') requiredFile: boolean;
    
    @Input('saveError') saveError: (error: ErrorMessage) => void;

    @Output('accreditationFile') accreditationFile: EventEmitter<
        File
    > = new EventEmitter();

    groupFields: Map<string, FormFieldGroup> = new Map();
    modalVisibility = false;

    constructor(
        private utilityService: UtilityService,
        public formStore: FormStoreService
    ) {}

    ngOnInit() {}

    createForm(accreditationModel: Accreditation): void {
        this.groupFields = new Map();
        this.groupFields.set('accreditation', {
            panel: false,
            panelHead: null,
            fields: new Map()
                .set('name', {
                    type: 'text',
                    label: 'CONFERENCE.JOIN.MODAL.FORM.NAME',
                    control: new FormControl(accreditationModel.name, [
                        Validators.required
                    ]),
                    errorLabels: [ErrorLabelConstants.REQUIRED],
                    size: '12|6|6'
                })
                .set('surname', {
                    type: 'text',
                    label: 'CONFERENCE.JOIN.MODAL.FORM.SURNAME',
                    control: new FormControl(accreditationModel.surname, [
                        Validators.required
                    ]),
                    errorLabels: [ErrorLabelConstants.REQUIRED],
                    size: '12|6|6'
                })
                .set('fiscalCode', {
                    type: 'text',
                    label: 'CONFERENCE.JOIN.MODAL.FORM.FISCAL_CODE',
                    pattern:
                        '^[a-zA-Z]{6}[0-9]{2}[a-zA-Z][0-9]{2}[a-zA-Z][0-9]{3}[a-zA-Z]$',
                    control: new FormControl(accreditationModel.fiscalCode, [
                        Validators.required
                    ]),
                    errorLabels: [
                        ErrorLabelConstants.REQUIRED,
                        ErrorLabelConstants.REGEX('ERROR.CF.INVALID')
                    ],
                    size: '12|6|6'
                })
                .set('profile', {
                    type: 'select',
                    label: 'CONFERENCE.JOIN.MODAL.FORM.PROFILE',
                    size: '12|6|6',
                    options: this.utilityService.getAccreditationRoles(),
                    control: new FormControl(
                        environment.defaultComboBox.conference.participants.users.profile,
                        [Validators.required]
                    ),
                    errorLabels: [ErrorLabelConstants.REQUIRED]
                })
                .set('email', {
                    type: 'text',
                    pattern: '^[a-z0-9._%+-]+@[a-z0-9.-]+.[a-z]{2,4}$',
                    label: 'CONFERENCE.JOIN.MODAL.FORM.EMAIL',
                    control: new FormControl(accreditationModel.email, [
                        Validators.required
                    ]),
                    errorLabels: [
                        ErrorLabelConstants.REQUIRED,
                        ErrorLabelConstants.REGEX('ERROR.EMAIL.INVALID')
                    ],
                    size: '12|6|6'
                })
                .set('file', {
                    type: 'file',
                    onSelect: this.upload.bind(this, accreditationModel),
                    hideSubmitBtn: true
                    // control: new FormControl(accreditationModel.accreditationFile, [Validators.required]),
                })
        });
    }

    upload(accreditationModel: Accreditation, file?: File): void {
        if (file === null) {
            this.formStore.valid = this.requiredFile; //TODO rendere obbligatorio per tutte le tipologie, tranne pre-istruttoria
        } else {
            this.formStore.valid = true;
            this.accreditationFile.emit(file);
            accreditationModel.setFile(file);
        }
    }

    openModal(): void {
        this.modalVisibility = true;
        this.modal.open();
    }

    close(): void {
        this.modalVisibility = false;
        this.modal.close();
    }
}
