import {
    Component,
    ViewEncapsulation,
    EventEmitter,
    Output
} from '@angular/core';
import { Observable } from 'rxjs';

import { HeaderService, LoaderService, SectionLoading } from '@app/core';
import { shareReplay, take } from 'rxjs/operators';

@Component({
    selector: 'eng-header',
    templateUrl: './eng-header.component.html',
    styleUrls: ['./eng-header.component.scss'],
    encapsulation: ViewEncapsulation.None
})
export class EngHeaderComponent {
    @Output('logout') logout: EventEmitter<void> = new EventEmitter();
    @Output('login') login: EventEmitter<void> = new EventEmitter();
    @Output('showMenu') showMenu: EventEmitter<void> = new EventEmitter();

    user$ = this.headerService.user$.pipe(
        shareReplay(1),
        take(1)
    );

    constructor(
        private loaderService: LoaderService,
        private headerService: HeaderService
    ) {}

    get loading$(): Observable<boolean> {
        return this.loaderService.getLoading$(SectionLoading.ALL_CONTENT);
    }

    get showUser(): boolean {
        return this.headerService.showUser;
    }
}
